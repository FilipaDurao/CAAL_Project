797 template<typename T>
798 Node<T> * Graph<T>::dijkstra_heap(Node<T> * startNode, Node<T> * endNode) {
799 
800  vector<Node<T> *> path = { };
801 
802  for (auto it = this->nodes.begin(); it != this->nodes.end(); it++) {
803  (*it)->setDistance(DBL_MAX);
804  (*it)->clearLastNode();
805  }
806 
807  startNode->setDistance(0);
808  path.push_back(startNode);
809 
810 //making the heap, since it only has one element does not need the function
811  make_heap(path.begin(), path.end());
812 
813  Node<T> * v;
814  Node<T> * w;
815  double new_distance;
816  double old_distance;
817 
818  while (!path.empty()) {
819 
820  //the miminum value is always in the top
821  v = path.front();
822 
823  //putting the min value (considered the max since we swap the operator) in the back
824  pop_heap(path.begin(), path.end());
825 
826  //removing it
827  path.pop_back();
828 
829  for (auto it = v->getEdges().begin(); it != v->getEdges().end(); it++) {
830 
831  w = it->getDestiny();
832  new_distance = v->getDistance() + it->getWeight();
833  old_distance = w->getDistance();
834 
835  if (old_distance > new_distance) {
836 
837  w->setDistance(new_distance);
838  w->setLastNode(v);
839 
840  if (old_distance == DBL_MAX) { //aka is not in the path
841  path.push_back(w);
842  }
843 
844  make_heap(path.begin(), path.end(), compareDistance<T>());
845 
846  }
847  }
848  }
849 
850  return endNode;
851 }

862 template<class T>
863 Node<T> * Graph<T>::A_Star(Node<T> * startNode, Node<T> * endNode) {
864 
865  //initial setup to compare by distance
866  Node<T>::sortByDistance = true;
867 
868  for (auto it = this->nodes.begin(); it != this->nodes.end(); it++) {
869  (*it)->setDistance(DBL_MAX);
870  (*it)->clearLastNode();
871  (*it)->setVisited(false);
872  (*it)->setLastConnection("NOT");
873  (*it)->setPrice(0);
874  }
875 
876  startNode->setDistance(0);
877  startNode->setLastConnection("FIRST");
878  MutablePriorityQueue<Node<T> > q;
879  q.insert(startNode);
880 
881  Node<T> * v;
882  Node<T> * w;
883  double old_distance;
884  double new_distance;
885 
886  while (!q.empty()) {
887 
888  v = q.extractMin();
889 
890  if (v->getId() == endNode->getId()) {
891  endNode->setDistance(
892  endNode->getDistance()
893  + startNode->euclidianDistance(endNode));
894  break;
895 
896  }
897 
898  for (auto it = v->getEdges().begin(); it != v->getEdges().end(); it++) {
899 
900  w = it->getDestiny();
901  old_distance = w->getDistance();
902  new_distance = (v->getDistance() + it->getWeight())
903  - v->euclidianDistance(endNode)
904  + w->euclidianDistance(endNode);
905 
906  if (old_distance > new_distance) {
907 
908  /*updating the prices
909  * not important for the queue since it's taking distance as the operator
910  */
911  if (v->getLastConnection() != it->getEdgeConnection())
912  w->setPrice(v->getPrice() + it->getPriceWeight());
913 
914  else
915  w->setPrice(v->getPrice());
916 
917  /*
918  * adding the transbord time if he changed the type of vehicle
919  * ignoring walking
920  */
921  if (v->getLastTypeConnection() != it->getType()
922  && v->getLastTypeConnection() != WALK
923  && it->getType() != WALK) {
924  new_distance += v->getTransbordTime();
925  }
926 
927  w->setDistance(new_distance);
928  w->setLastNode(v);
929  w->setLastConnection(it->getEdgeConnection());
930  if (!w->getVisited())
931  q.insert(w);
932  else
933  q.decreaseKey(w);
934 
935  w->setVisited(true);
936 
937  }
938  }
939  }
940 
941  return endNode;
942 
943 }
